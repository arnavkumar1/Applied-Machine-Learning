def mylinridgereg(x,y,l):
    w = np.dot(inv(np.dot(np.transpose(x), x) + l * np.identity(len(x[0]))) , np.dot(np.transpose(x), y))
    return w

def meansquarederr(T, Tdash):
    error = 0
    for i in range(len(T)):
        error += pow((T[i] - Tdash[i]), 2)
    return error/len(T)
